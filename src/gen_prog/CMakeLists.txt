set(target gen_prog)



#######################################################################################################################
#######################################################################################################################
#
# sources definition
#
#######################################################################################################################
#######################################################################################################################

set(${target}_INCLUDE_DIR ${CMAKE_SOURCE_DIR}/include/${target})

set(
    ${target}_HEADERS
    ${${target}_INCLUDE_DIR}/environ.hpp
)



#######################################################################################################################
#######################################################################################################################
#
# build command
#
#######################################################################################################################
#######################################################################################################################


cmut_define_output_dirs()
add_library(
    ${target}
    ${${target}_HEADERS}
)
set_target_properties(${target}
    PROPERTIES
        LINKER_LANGUAGE CXX
    )


cmut__target__generate_export_header(${target})
cmut__target__set_library_version(${target} ${PROJECT_VERSION})



cmut__build__msvc__secure_no_warning(${target})
cmut__build__msvc__nominmax(${target})

target_include_directories(
    ${target}
    PUBLIC
        "$<BUILD_INTERFACE:${CMAKE_SOURCE_DIR}/include>"
        )

#target_link_libraries(
#    ${target}
#)


#######################################################################################################################
#######################################################################################################################
#
# install command
#
#######################################################################################################################
#######################################################################################################################

cmut__install__install_library(${target})
